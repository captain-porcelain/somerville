Project one point on a unit sphere onto a plane.

(defn to-plane
  [p]
  (point/point (/ (:x p) (- 1 (:z p))) (/ (:y p) (- 1 (:z p)))))

Project a point from the plane onto a unit sphere.

(defn to-sphere
  [p]
  (let [lower (+ 1 (* (:x p) (:x p)) (* (:y p) (:y p)))]
    (point/point
      (/ (* 2 (:x p)) lower)
      (/ (* 2 (:y p)) lower)
      (/ (+ -1 (* (:x p) (:x p)) (* (:y p) (:y p))) lower))))
